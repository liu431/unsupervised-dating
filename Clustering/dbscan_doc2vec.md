modal = function(vect, percent = FALSE) {
  library(tidyverse)
  modal_val = vect %>% unlist() %>% table() %>% .[. == max(.)] %>% names()
  if (percent) {
    return(vect %>% unlist() %>% .[. == modal_val] %>% (function(x) length(x) / length(vect)))
    }
  modal_val
}
